KLPLevels = document.getElementsByClassName("KLPLevel");
KLPLayer = document.getElementById("KLP");
KLPLayer.addEventListener("click", (event) => {
    if (KLPLayer.checked === true){
        for (let i of KLPLevels) {
        let a =SelectedLevels.indexOf(i.value);
        if (a !== -1) {
            SelectedLevels.splice(SelectedLevels.indexOf(a.value), 1);
        }
        }    
        for (let i of KLPLevels) {
        i.checked = true;
        SelectedLevels.push(i.value);
        }
    }
    else{
        for (let i of KLPLevels) {
        i.checked = false;
        SelectedLevels.splice(SelectedLevels.indexOf(i.value))
        }
    }
    console.log(SelectedLevels)
})

document.getElementById("KAKOpen").addEventListener("click", (event) => {
    if (document.getElementById("KAKOpen").checked === true){
        for (let i of document.getElementsByClassName("KAKLabel")) {
        i.style.visibility = "visible"
    }
    for (let i of document.getElementsByClassName("KAKLevel")) {
        i.style.visibility = "visible"
    }
    document.getElementById("KAKOpenLabel").textContent = "<"
    }
    else {
        for (let i of document.getElementsByClassName("KAKLabel")) {
        i.style.visibility = "hidden"
    }
    for (let i of document.getElementsByClassName("KAKLevel")) {
        i.style.visibility = "hidden"
    }
    document.getElementById("KAKOpenLabel").textContent = ">"
    }
})
